rainbow
      
        #rainbow {
          animation-name: rainbow;
          animation-duration: 4s;
          animation-iteration-count: 10;
        }
      
        @keyframes rainbow {
          0% {
            background-color: blue;
          }
          50% {
            background-color: green;
          }
          100% {
            background-color: yellow;
          }
        }
